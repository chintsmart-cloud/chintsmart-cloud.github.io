0 开关¶

请求示例

{
    "category_id": 0,
    "properties": {
        // no required 进入睡眠等待时间
        "enterSleepAfterTime": {
            "min": "0",
            "max": "65535"
            // "unitName": "秒",
            // "step": "1",
            // "defaultValue": "0"
        },

        // no required 开关
        "powerstate": true,

        // no required 激活设备唤醒时间
        "activeWakeupTime": {
            "min": "0",
            "max": "65535",
            // "unit": "s",
            // "unitName": "秒",
            // "step": "1",
            // "defaultValue": "5"
        },

        // no required 设备端上定时
        "device_timer": {
            // 定时开始时间
            "timer": "",
            "enable": true,
            // 定时执行的操作
            "targets": "",
            "time_zone": {
                "min": "-43200",
                "max": "50400",
                "unit": "s",
                "unitName": "秒",
                "step": "1",
                "defaultValue": "0"
            },
            // "0": "未配置",
            // "1": "倒计时",
            // "2": "普通定时",
            // "3": "周期循环定时",
            // "4": "随机定时"
            "type": 0,
            "end_time": "",
            "run_time": {
                "min": "0",
                "max": "1440",
                "unit": "m",
                "unitName": "分钟",
                "step": "1",
                "defaultValue": "0"
            },
            "sleep_time": {
                "min": "0",
                "max": "1440",
                "unit": "m",
                "unitName": "分钟",
                "step": "1",
                "defaultValue": "0"
            }
        },

        // no required 模式
        // "3": "阅读模式",
        // "8": "起床模式",
        // "230": "节能模式",
        // "378": "休闲模式",
        // "395": "会客模式",
        // "751": "全开模式",
        // "982": "回家模式",
        // "983": "离家模式",
        // "984": "安防模式",
        // "985": "解除安防"
        "mode": 3,

        // no required 发射功率
        "transmissionPower": {
            "min": "0",
            "max": "100",
            "unit": "none",
            "unitName": "无",
            "step": "1",
            "defaultValue": "0"
        }



    }
}

请求参数说明

参数名

必选

类型

说明

access_token

是

string

访问token

device_id

是

string

设备的唯一id

device_secret

否

string

设备密钥

mac_addr

是

string

设备物理地址

sdk_version

是

string

SDK版本

firmware_version

是

string

设备固件版本

nick_name

否

string

设备昵称

net_addr

否

string

设备网络地址

status

是

integer

设备状态, 0:未激活; 1:在线; 2:离线; 3:禁用; 4:已激活

group_id

否

integer

要查询的空间，如家ID、房间ID等

is_sub_device

否

boolean

是否为子设备。归属在某一个网关设备下的设备表示为子设备。 取值为：true （表示获取子设备列表）；false （表示获取直连设备列表）

owned

否

integer

用户获取设备的方式, 0:被分享的设备; 1:拥有的设备

product_id

是

string

产品的唯一id

category_id

是

string

品类id, 和物模型配对, 参考 公共参数说明-物模型

        "PowerSwitch": true,

        "RMSVoltage": "xxxx",
        "CountDown": {
            "IsRunning": true,
            "TimeLeft": {
                "min": 1,
                "max": 86399,
                "PowerSwitch": true,
                "Timestamp": 0
            }
        }
    }
}
